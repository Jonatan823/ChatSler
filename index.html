function slerize(text) { // Límite estricto de 25-30 caracteres para máxima verticalidad const LIMIT = 25; let words = text.split(/\s+/); let lines = []; let currentLine = ""; words.forEach((word, index) => { // Si la palabra cabe en la línea actual if ((currentLine + " " + word).trim().length <= LIMIT) { currentLine += (currentLine === "" ? "" : " ") + word; } else { // Aplicar guion de salida en línea M lines.push(currentLine.trim() + "-"); // Aplicar guion de entrada en línea C (Reciprocidad) currentLine = "-" + word; } }); // Agregar la última línea sin guion final if (currentLine !== "") { lines.push(currentLine.trim()); } return lines.join('\n'); }

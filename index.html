function motorSLER(texto) { return texto.split(" ").map(palabra => { // Solo procesamos palabras largas para evitar romper el ritmo if (palabra.length > 5) { let mitad = Math.ceil(palabra.length / 2); let parte1 = palabra.substring(0, mitad); let parte2 = palabra.substring(mitad); // Estructura Zig-Zag: // Línea M (Impar): Guion al final para inducir completitud. // Línea C (Par): Guion al inicio para advertir inconclusividad. return `
${parte1}-
-${parte2}
`; } return `
${palabra}
`; }).join(""); }
